// define functions, calculations
double celtoFar(double celsius) {
	return(celsius * 9.0 / 5.0) + 32;
}
double fartoCel (double fahrenheit) {
	return(fahrenheit - 32) * 5.0 / 9.0;
}
double keltoCel (double kelvin) {
	return kelvin - 273.15;
}
double celtoKel (double celsius) {
	return celsius + 273.15;
}

int calculator(int num1, int num2, int operation) {
	switch(operation)
	{	// FORMULAS
		case 1: return num1 + num2;
		case 2: return num1 - num2;
		case 3: return (num2 != 0) ? num1 / num2 : 0;
		case 4: return num1 * num2;
		default: return 0;
	}
}
